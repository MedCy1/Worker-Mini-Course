📄 Le code

// main.js — thread principal
const worker = new Worker('worker.js');

// Envoyer une tâche au worker
worker.postMessage({ limit: 10_000_000 });

// Recevoir le résultat sans bloquer
worker.onmessage = (e) => {
  console.log('Primes found:', e.data.count);
  console.log('Time:', e.data.duration + 'ms');
};

// L'UI continue de répondre pendant ce temps !
// worker.js — même algorithme, thread séparé
self.onmessage = (e) => {
  const { limit } = e.data; // 8 000 000
  const t0 = performance.now();
  let count = 0;

  // Division par essai — identique au main thread
  for (let n = 2; n <= limit; n++) {
    let prime = true;
    for (let d = 2; d * d <= n; d++) {
      if (n % d === 0) { prime = false; break; }
    }
    if (prime) count++;
    // Progression envoyée tous les 200 000 nombres
    if (n % 200_000 === 0)
      self.postMessage({ progress: n / limit });
  }

  // Résultat final — UI n'a jamais été bloquée
  self.postMessage({ count, duration: (performance.now() - t0).toFixed(1) });
};
// Sans worker — division par essai naïve dans le main thread
const LIMIT = 8_000_000;
const t0 = performance.now();
let count = 0;

// O(n√n) — intentionnellement lent pour démontrer le freeze
outer: for (let n = 2; n <= LIMIT; n++) {
  for (let d = 2; d * d <= n; d++) {
    if (n % d === 0) continue outer;
  }
  count++;
}
