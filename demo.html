Le code

// main.js - main thread
const worker = new Worker('worker.js');

// Send a task to the worker
worker.postMessage({ limit: 10_000_000 });

// Receive the result without blocking
worker.onmessage = (e) => {
  console.log('Primes found:', e.data.count);
  console.log('Time:', e.data.duration + 'ms');
};

// The UI keeps responding during this time!
// worker.js - same algorithm, separate thread
self.onmessage = (e) => {
  const { limit } = e.data; // 8 000 000
  const t0 = performance.now();
  let count = 0;

  // Trial division - identical to the main thread version
  for (let n = 2; n <= limit; n++) {
    let prime = true;
    for (let d = 2; d * d <= n; d++) {
      if (n % d === 0) { prime = false; break; }
    }
    if (prime) count++;
    // Send progress every 200,000 numbers
    if (n % 200_000 === 0)
      self.postMessage({ progress: n / limit });
  }

  // Final result - UI was never blocked
  self.postMessage({ count, duration: (performance.now() - t0).toFixed(1) });
};
// Without worker - naive trial division in the main thread
const LIMIT = 8_000_000;
const t0 = performance.now();
let count = 0;

// O(n√n) - intentionally slow to demonstrate the freeze
outer: for (let n = 2; n <= LIMIT; n++) {
  for (let d = 2; d * d <= n; d++) {
    if (n % d === 0) continue outer;
  }
  count++;
}

// The browser was ENTIRELY frozen for ~2-4s
// No clicks, no animations, no rendering possible
